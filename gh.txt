name: Create MySQL Table

on: [push]

jobs:
  build:
    runs-on: ubuntu-latest

    steps:
    - name: Checkout code
      uses: actions/checkout@v2

    - name: Set up Python
      uses: actions/setup-python@v2
      with:
        python-version: '3.x'

    - name: Install dependencies
      run: |
        python -m pip install --upgrade pip
        pip install mysql-connector-python

    - name: Create MySQL database and table
      run: |
        import mysql.connector
        db = mysql.connector.connect(
          host="${{ secrets.RDS_ENDPOINT }}",
          user="${{ secrets.RDS_USERNAME }}",
          password="${{ secrets.RDS_PASSWORD }}"
        )
        cursor = db.cursor()
        cursor.execute("CREATE DATABASE IF NOT EXISTS your_database_name;")
        db.commit()
        db.close()

        db = mysql.connector.connect(
          host="${{ secrets.RDS_ENDPOINT }}",
          user="${{ secrets.RDS_USERNAME }}",
          password="${{ secrets.RDS_PASSWORD }}",
          database="your_database_name"
        )
        cursor = db.cursor()
        cursor.execute("""
          CREATE TABLE your_table_name (
            id INT AUTO_INCREMENT PRIMARY KEY,
            name VARCHAR(40) NOT NULL,
            email VARCHAR(40)
          )
        """)
        db.close()
      shell: python
